    <div class="pillar">
      <div class="pillar-icon"><i class="fa-brands fa-docker"></i></div>

    </div>
    <div class="pillar">
      <div class="pillar-icon"><i class="fa-solid fa-cube"></i></div>
      <h3>Simple</h3>
      <div class="pillar-text">One Helm chart. Same config on minikube and EKS Auto Mode. 12 lines of YAML to define your pipeline.</div>
    </div>
